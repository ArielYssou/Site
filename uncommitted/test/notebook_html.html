def make_windows(df, start, end, window_size, target_size):
    '''
    DOCSTRING
    '''
    data = []
    target = []

    if end == -1:
        len(df) - (target_size - 1)
    
    # ens == -1 should be interpreted as the end of the data
    for i in range(start + window_size, end):
        indices = range(i - window_size, i)
        
        # Reshape data from (history_size,) to (history_size, 1)
        data.append(np.reshape(df[indices], (window_size, 1)))
        target.append(df[i + (target_size - 1)])
        
    return np.array(data), np.array(target)

def show_prediction_window(data, targets, predictions):
    fig, ax = plt.subplots(1)
    time_data = list(range(-len(data), 0))
    time_target = list(range(0, len(targets)))
    time_pred = list(range(0, len(predictions)))
    
    ax.plot(time_data, data,
             marker = '.', c = 'skyblue', label="History")
    ax.plot(time_target, targets,
             marker = 'x', c = 'red', label="True future")
    ax.plot(time_pred, predictions,
             marker = 'o', c = 'green', label="Predictions")
    plt.show()
